  ```python
      numbered = '\n'.join(f'{i + 1}. {t}' for i, t in enumerate(titles))

      except Exception as e:
  ```

  ```python
      """Single Gemini call to analyse all watchlist tickers. Returns dict keyed by ticker."""
      if not items:
          return {}

      p     = PROMPTS[LANG]
      parts = []
      for item in items:
          part = (

              + p['watchlist_price_label'].format(

          )

              part += p['watchlist_news_label']
              part += '\n'.join(f'  - {t}' for t in item['news_titles'][:3]) + '\n'
          parts.append(part)

      context_hint = p['watchlist_ctx_morning'] if run_type == 'morning' else p['watchlist_ctx_evening']
      ticker_list  = json.dumps([i['ticker'] for i in items], ensure_ascii=False)
      prompt = (
  ```

  ```python
  def analyze_portfolio_risk(portfolio_data, news_entries, models=None):
      """Cross-reference portfolio holdings with recent news to surface relevant alerts."""
      if not portfolio_data or not news_entries:
          return []

      positions = [p for p in portfolio_data.get('positions', []) if not p.get('error')]
      if not positions:
          return []

      holdings_text = '\n'.join(f"  - {pos['ticker']} ({pos['name']})" for pos in positions)
      titles        = [getattr(e, 'title', str(e)) for e in news_entries[:20]]
      if not titles:
          return []
      news_text   = '\n'.join(f'{i+1}. {t}' for i, t in enumerate(titles))
      ticker_list = json.dumps([p['ticker'] for p in positions], ensure_ascii=False)

      prompt = PROMPTS[LANG]['risk'].format(
          holdings=holdings_text, n=len(titles), news=news_text, ticker_list=ticker_list
      )

      try:
          raw   = chat_with_gemini(prompt, models=models or WATCHLIST_MODEL_CHAIN)
          match = re.search(r'\{.*\}', raw, re.DOTALL)
          if match:
              alerts = json.loads(match.group()).get('alerts', [])
              valid_tickers = {p['ticker'] for p in positions}
              result = []
              for a in alerts:
                  if not isinstance(a, dict):
                      continue
                  ticker = a.get('ticker', '')
                  alert  = a.get('alert', '').strip()
                  if ticker in valid_tickers and alert:
                      name = next((p['name'] for p in positions if p['ticker'] == ticker), ticker)
                      result.append({'ticker': ticker, 'name': name, 'alert': alert})
              return result
      except Exception as e:
          print(f"  analyze_portfolio_risk failed: {e}")
      return []
  ```

## Task 6: Create .github/workflows/demo.yml

**Files:**
- Create: `.github/workflows/demo.yml`

- [ ] **Step 1: Write the workflow file**

  ```yaml
  name: Demo Daily Digest

  on:
    workflow_dispatch:
    schedule:
      - cron: '30 22 * * *'  # UTC 22:30 = JST 07:30

  permissions:
    contents: read
    pages: write
    id-token: write

  jobs:
    demo-digest:
      runs-on: ubuntu-latest
      steps:
        - uses: actions/checkout@v4

        - uses: actions/setup-python@v5
          with:
            python-version: '3.11'

        - name: Install dependencies
          run: pip install -r requirements.txt

        - name: Run demo digest
          env:
            GEMINI_API_KEY: ${{ secrets.GEMINI_API_KEY }}
          run: python main.py --demo --test

        - name: Upload HTML artifact
          uses: actions/upload-artifact@v4
          with:
            name: demo-digest-${{ github.run_number }}
            path: demo/output/daily.html
            retention-days: 30

        - name: Upload Pages artifact
          uses: actions/upload-pages-artifact@v3
          with:
            path: demo/output/

    deploy-pages:
      needs: demo-digest
      runs-on: ubuntu-latest
      environment:
        name: github-pages
        url: ${{ steps.deployment.outputs.page_url }}
      steps:
        - id: deployment
          uses: actions/deploy-pages@v4
  ```
